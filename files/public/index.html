// 修改用户名密码弹窗逻辑 const showChangeCredBtn = document.getElementById('show-change-cred'); const changeCredModal = document.getElementById('change-cred-modal'); const closeChangeCredBtn = document.getElementById('close-change-cred'); const changeCredForm = document.getElementById('change-cred-form'); const oldUserI = document.getElementById('old-user'); const oldPassI = document.getElementById('old-pass'); const newUserI = document.getElementById('new-user'); const newPassI = document.getElementById('new-pass'); const credStatus = document.getElementById('cred-status'); showChangeCredBtn.onclick = () => { changeCredModal.classList.remove('hidden'); credStatus.textContent = ''; oldUserI.value = userI.value; oldPassI.value = passI.value; newUserI.value = ''; newPassI.value = ''; }; closeChangeCredBtn.onclick = () => { changeCredModal.classList.add('hidden'); }; changeCredModal.onclick = (e) => { if (e.target === changeCredModal) changeCredModal.classList.add('hidden'); }; if (changeCredForm) { changeCredForm.onsubmit = async (e) => { e.preventDefault(); credStatus.textContent = ''; const oldUser = oldUserI.value.trim(); const oldPass = oldPassI.value.trim(); const newUser = newUserI.value.trim(); const newPass = newPassI.value.trim(); if (!oldUser || !oldPass || !newUser || !newPass) { credStatus.textContent = '所有字段均不能为空'; return; } credStatus.textContent = '提交中...'; try { const resp = await fetch('/change-cred', { method: 'POST', headers: { 'Authorization': 'Basic ' + btoa(oldUser + ':' + oldPass), 'Content-Type': 'application/json' }, body: JSON.stringify({ user: newUser, pass: newPass }) }); const data = await resp.json(); if (data.ok) { credStatus.textContent = '修改成功，请用新用户名密码登录'; setTimeout(() => { changeCredModal.classList.add('hidden'); localStorage.removeItem('rb-ui-auth'); location.reload(); }, 1200); } else { credStatus.textContent = data.error || '修改失败'; } } catch (err) { credStatus.textContent = '请求失败'; } }; }
